TITLE first Example Data definition
INNcLUDE Irvine32.inc
.data
var1	byte	10h
array1	byte	25h,90h,3 dup(95h)
var2	word	1324h
array2	word	3 dup (950h)
var3	dword	588955h
array3	dword	2 dup(3A54BC32h)


.code
main PROC
callclrscr

mov ebx, offset var1
mov ecx ,offset array1
mov edx, offset var2
mov edi, offset array2
mov eax, offset var3
mov esi, offset array3

mov al, DS:[404000h]
mov bx, DS:[404000h]
mov ecx, DS:[404000h]
call dumpregs
	exit
main ENDP
END main
